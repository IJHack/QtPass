100void tst_executor::gpgSupportsEd25519() {
101 QString output;
102 QString err;
103 int gpgVersionExitCode =
104 Executor::executeBlocking("gpg", {"--version"}, QString(), &output, &err);
105
106 if (gpgVersionExitCode != 0) {
107 QSKIP("GPG not available");
108 }
109
110 QVERIFY2(output.contains("gpg") || output.contains("GnuPG"),
111 "GPG version output should be present");
112
113 QRegularExpression versionRegex(R"(gpg \‍(GnuPG\) (\d+)\.(\d+))");
114 QRegularExpressionMatch match = versionRegex.match(output);
115 if (!match.hasMatch()) {
116 qWarning() << "GPG output:" << output;
117 qWarning() << "GPG stderr:" << err;
118 QSKIP("Could not parse GPG version");
119 }
120
121 int major = match.captured(1).toInt();
122 int minor = match.captured(2).toInt();
123 bool expectEd25519 = major > 2 || (major == 2 && minor >= 1);
124
125 bool result = Pass::gpgSupportsEd25519();
126 if (expectEd25519) {
127 if (!result) {
128 qWarning() << "Pass::gpgSupportsEd25519() returned false for GPG" << major
129 << minor;
130 QSKIP("QtPassSettings not properly initialized");
131 }
132 QVERIFY2(
133 result == true,
134 qPrintable(
135 QString("GPG %1.%2 should support Ed25519").arg(major).arg(minor)));
136 } else {
137 QVERIFY2(result == false,
138 qPrintable(QString("GPG %1.%2 should not support Ed25519")
139 .arg(major)
140 .arg(minor)));
141 }
142}

144void tst_executor::getDefaultKeyTemplate() {
145 bool ed25519Supported = Pass::gpgSupportsEd25519();
146 QString templateStr = Pass::getDefaultKeyTemplate();
147 QVERIFY2(!templateStr.isEmpty(), "Template should not be empty");
148 QVERIFY2(templateStr.contains("Key-Type:"),
149 "Template should contain Key-Type");
150 QVERIFY2(templateStr.contains("%echo done"),
151 "Template should contain done marker");
152 if (ed25519Supported) {
153 QVERIFY2(templateStr.contains("ed25519") || templateStr.contains("EdDSA"),
154 "Template should be the Ed25519 variant when gpgSupportsEd25519() "
155 "is true");
156 } else {
157 QVERIFY2(templateStr.contains("RSA"), "Template should be the RSA fallback "
158 "when gpgSupportsEd25519() is false");
159 }
160}
