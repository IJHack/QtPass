212 void MainWindow::mountWebDav() {
213 #ifdef Q_OS_WIN
214  char dst[20] = {0};
215  NETRESOURCEA netres;
216  memset(&netres, 0, sizeof(netres));
217  netres.dwType = RESOURCETYPE_DISK;
218  netres.lpLocalName = 0;
219  netres.lpRemoteName = QtPassSettings::getWebDavUrl().toUtf8().data();
220  DWORD size = sizeof(dst);
221  DWORD r = WNetUseConnectionA(
222  reinterpret_cast<HWND>(effectiveWinId()), &netres,
223  QtPassSettings::getWebDavPassword().toUtf8().constData(),
224  QtPassSettings::getWebDavUser().toUtf8().constData(),
225  CONNECT_TEMPORARY | CONNECT_INTERACTIVE | CONNECT_REDIRECT, dst, &size,
226  0);
227  if (r == NO_ERROR) {
228  QtPassSettings::setPassStore(dst);
229  } else {
230  char message[256] = {0};
231  FormatMessageA(FORMAT_MESSAGE_FROM_SYSTEM, 0, r, 0, message,
232  sizeof(message), 0);
233  ui->textBrowser->setTextColor(Qt::red);
234  ui->textBrowser->setText(tr("Failed to connect WebDAV:\n") + message +
235  " (0x" + QString::number(r, 16) + ")");
236  ui->textBrowser->setTextColor(Qt::black);
237  }
238 #else
239  fusedav.start("fusedav -o nonempty -u \"" + QtPassSettings::getWebDavUser() +
240  "\" " + QtPassSettings::getWebDavUrl() + " \"" +
241  QtPassSettings::getPassStore() + '"');
242  fusedav.waitForStarted();
243  if (fusedav.state() == QProcess::Running) {
244  QString pwd = QtPassSettings::getWebDavPassword();
245  bool ok = true;
246  if (pwd.isEmpty()) {
247  pwd = QInputDialog::getText(this, tr("QtPass WebDAV password"),
248  tr("Enter password to connect to WebDAV:"),
249  QLineEdit::Password, "", &ok);
250  }
251  if (ok && !pwd.isEmpty()) {
252  fusedav.write(pwd.toUtf8() + '\n');
253  fusedav.closeWriteChannel();
254  fusedav.waitForFinished(2000);
255  } else {
256  fusedav.terminate();
257  }
258  }
259  QString error = fusedav.readAllStandardError();
260  int prompt = error.indexOf("Password:");
261  if (prompt >= 0)
262  error.remove(0, prompt + 10);
263  if (fusedav.state() != QProcess::Running)
264  error = tr("fusedav exited unexpectedly\n") + error;
265  if (error.size() > 0) {
266  ui->textBrowser->setTextColor(Qt::red);
267  ui->textBrowser->setText(
268  tr("Failed to start fusedav to connect WebDAV:\n") + error);
269  ui->textBrowser->setTextColor(Qt::black);
270  }
271 #endif
272 }

941 void MainWindow::onUsers() {
942  QList<UserInfo> users = QtPassSettings::getPass()->listKeys();
943  if (users.size() == 0) {
944  QMessageBox::critical(this, tr("Can not get key list"),
945  tr("Unable to get list of available gpg keys"));
946  return;
947  }
948  QList<UserInfo> secret_keys = QtPassSettings::getPass()->listKeys("", true);
949  foreach (const UserInfo &sec, secret_keys) {
950  for (QList<UserInfo>::iterator it = users.begin(); it != users.end(); ++it)
951  if (sec.key_id == it->key_id)
952  it->have_secret = true;
953  }
954  QList<UserInfo> selected_users;
955  QString dir =
956  currentDir.isEmpty()
957  ? Util::getDir(ui->treeView->currentIndex(), false, model, proxyModel)
958  : currentDir;
959  int count = 0;
960  QString recipients = QtPassSettings::getPass()->getRecipientString(
961  dir.isEmpty() ? "" : dir, " ", &count);
962  if (!recipients.isEmpty())
963  selected_users = QtPassSettings::getPass()->listKeys(recipients);
964  foreach (const UserInfo &sel, selected_users) {
965  for (QList<UserInfo>::iterator it = users.begin(); it != users.end(); ++it)
966  if (sel.key_id == it->key_id)
967  it->enabled = true;
968  }
969  if (count > selected_users.size()) {
970  // Some keys seem missing from keyring, add them separately
971  QStringList recipients =
972  QtPassSettings::getPass()->getRecipientList(dir.isEmpty() ? "" : dir);
973  foreach (const QString recipient, recipients) {
974  if (QtPassSettings::getPass()->listKeys(recipient).size() < 1) {
975  UserInfo i;
976  i.enabled = true;
977  i.key_id = recipient;
978  i.name = " ?? " + tr("Key not found in keyring");
979  users.append(i);
980  }
981  }
982  }
983  UsersDialog d(this);
984  d.setUsers(&users);
985  if (!d.exec()) {
986  d.setUsers(NULL);
987  return;
988  }
989  d.setUsers(NULL);
990 
991  QtPassSettings::getPass()->Init(dir, users);
992 }
