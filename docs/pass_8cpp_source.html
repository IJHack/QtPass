66 parts.append(v);

68 *it = wslenvPrefix + parts.join(':');
69 }

123auto Pass::generatePassword(unsigned int length, const QString &charset)
124 -> QString {
125 QString passwd;
126 if (QtPassSettings::isUsePwgen()) {
127 // --secure goes first as it overrides --no-* otherwise
128 QStringList args;
129 args.append("-1");
130 if (!QtPassSettings::isLessRandom()) {
131 args.append("--secure");
132 }
133 args.append(QtPassSettings::isAvoidCapitals() ? "--no-capitalize"
134 : "--capitalize");
135 args.append(QtPassSettings::isAvoidNumbers() ? "--no-numerals"
136 : "--numerals");
137 if (QtPassSettings::isUseSymbols()) {
138 args.append("--symbols");
139 }
140 args.append(QString::number(length));
141 // executeBlocking returns 0 on success, non-zero on failure
142 if (Executor::executeBlocking(QtPassSettings::getPwgenExecutable(), args,
143 &passwd) == 0) {
144 static const QRegularExpression literalNewLines{"[\\n\\r]"};
145 passwd.remove(literalNewLines);
146 } else {
147 passwd.clear();
148#ifdef QT_DEBUG
149 qDebug() << __FILE__ << ":" << __LINE__ << "\t"
150 << "pwgen fail";
151#endif
152 // Error is already handled by clearing passwd; no need for critical
153 // signal here
154 }
155 } else {
156 // Validate charset - if CUSTOM is selected but chars are empty,
157 // fall back to ALLCHARS to prevent weak passwords (issue #780)
158 QString effectiveCharset = charset;
159 if (effectiveCharset.isEmpty()) {
160 effectiveCharset = QtPassSettings::getPasswordConfiguration()
161 .Characters[PasswordConfiguration::ALLCHARS];
162 }
163 if (effectiveCharset.length() > 0) {
164 passwd = generateRandomPassword(effectiveCharset, length);
165 } else {
166 emit critical(
167 tr("No characters chosen"),
168 tr("Can't generate password, there are no characters to choose from "
169 "set in the configuration!"));
170 }
171 }
172 return passwd;
173}

180bool Pass::gpgSupportsEd25519() {
181 QString out, err;
182 if (Executor::executeBlocking(QtPassSettings::getGpgExecutable(),
183 {"--version"}, &out, &err) != 0) {
184 return false;
185 }
186 QRegularExpression versionRegex(R"(gpg \‍(GnuPG\) (\d+)\.(\d+))");
187 QRegularExpressionMatch match = versionRegex.match(out);
188 if (!match.hasMatch()) {
189 return false;
190 }
191 int major = match.captured(1).toInt();
192 int minor = match.captured(2).toInt();
193 return major > 2 || (major == 2 && minor >= 1);
194}

287QStringList splitCommandCompat(const QString &command) {
288 QStringList result;
289 QString current;
290 bool inSingleQuote = false;
291 bool inDoubleQuote = false;
292 bool escaping = false;
293 for (QChar ch : command) {
294 if (escaping) {
295 current.append(ch);
296 escaping = false;
297 continue;
298 }
299 if (ch == '\\') {
300 escaping = true;
301 continue;
302 }
303 if (ch == '\'' && !inDoubleQuote) {
304 inSingleQuote = !inSingleQuote;
305 continue;
306 }
307 if (ch == '"' && !inSingleQuote) {
308 inDoubleQuote = !inDoubleQuote;
309 continue;
310 }
311 if (ch.isSpace() && !inSingleQuote && !inDoubleQuote) {
312 if (!current.isEmpty()) {
313 result.append(current);
314 current.clear();
315 }
316 continue;
317 }
318 current.append(ch);
319 }
320 if (escaping) {
321 current.append('\\');
322 }
323 if (!current.isEmpty()) {
324 result.append(current);
325 }
326 return result;
327}

345auto Pass::resolveGpgconfCommand(const QString &gpgPath)
346 -> ResolvedGpgconfCommand {
347 if (gpgPath.trimmed().isEmpty()) {
348 return {"gpgconf", {}};
349 }
350
351#if QT_VERSION >= QT_VERSION_CHECK(5, 15, 0)
352 QStringList parts = QProcess::splitCommand(gpgPath);
353#else
354 QStringList parts = splitCommandCompat(gpgPath);
355#endif
356
357 if (parts.isEmpty()) {
358 return {"gpgconf", {}};
359 }
360
361 const QString first = parts.first();
362 if (first.compare("wsl", Qt::CaseInsensitive) == 0 ||
363 first.compare("wsl.exe", Qt::CaseInsensitive) == 0) {
364 if (parts.size() >= 2 && parts.at(1).startsWith("sh")) {
365 return {"gpgconf", {}};
366 }
367 if (parts.size() >= 2 &&
368 QFileInfo(parts.last()).fileName().startsWith("gpg")) {
369 QString wslGpgconf = resolveWslGpgconfPath(parts.last());
370 parts.removeLast();
371 parts.append(wslGpgconf);
372 return {parts.first(), parts.mid(1)};
373 }
374 return {"gpgconf", {}};
375 }
376
377 if (!first.contains('/') && !first.contains('\\')) {
378 return {"gpgconf", {}};
379 }
380
381 QString gpgconfPath = findGpgconfInGpgDir(first);
382 if (!gpgconfPath.isEmpty()) {
383 return {gpgconfPath, {}};
384 }
385
386 return {"gpgconf", {}};
387}

393void Pass::GenerateGPGKeys(QString batch) {
394 // Kill any stale GPG agents that might be holding locks on the key database
395 // This helps avoid "database locked" timeouts during key generation
396 QString gpgPath = QtPassSettings::getGpgExecutable();
397 if (!gpgPath.isEmpty()) {
398 ResolvedGpgconfCommand resolvedGpgconf = resolveGpgconfCommand(gpgPath);
399 QStringList killArgs = resolvedGpgconf.arguments;
400 killArgs << "--kill";
401 killArgs << "gpg-agent";
402 // Use same environment as key generation to target correct gpg-agent
403 Executor::executeBlocking(env, resolvedGpgconf.program, killArgs);
404 }
405
406 executeWrapper(GPG_GENKEYS, gpgPath, {"--gen-key", "--no-tty", "--batch"},
407 std::move(batch));
408}

485auto gpgErrorMessage(const QString &err) -> QString {
486 // Machine-readable status tokens added by --status-fd 2
487 if (containsAny(err, {QStringLiteral("[GNUPG:] KEYEXPIRED"),
488 QStringLiteral("[GNUPG:] INV_RECP 5 ")}))
489 return QCoreApplication::translate(
490 "Pass", "Encryption failed: GPG key has expired. Please renew or "
491 "replace it.");
492 if (containsAny(err, {QStringLiteral("[GNUPG:] KEYREVOKED"),
493 QStringLiteral("[GNUPG:] INV_RECP 4 ")}))
494 return QCoreApplication::translate(
495 "Pass", "Encryption failed: GPG key has been revoked.");
496 if (containsAny(err, {QStringLiteral("[GNUPG:] NO_PUBKEY"),
497 QStringLiteral("[GNUPG:] INV_RECP")}))
498 return QCoreApplication::translate(
499 "Pass", "Encryption failed: recipient GPG key not found or invalid. "
500 "Check that the key ID in .gpg-id is correct and imported.");
501 if (err.contains(QStringLiteral("[GNUPG:] FAILURE")))
502 return QCoreApplication::translate(
503 "Pass", "Encryption failed. Check that your GPG key is valid.");
504
505 // Locale-dependent fallbacks
506 if (containsAnyCaseInsensitive(err, {QLatin1String("key has expired"),
507 QLatin1String("key expired")}))
508 return QCoreApplication::translate(
509 "Pass", "Encryption failed: GPG key has expired. Please renew or "
510 "replace it.");
511 if (containsAnyCaseInsensitive(err, {QLatin1String("key has been revoked"),
512 QLatin1String("revoked")}))
513 return QCoreApplication::translate(
514 "Pass", "Encryption failed: GPG key has been revoked.");
515 if (containsAnyCaseInsensitive(err, {QLatin1String("no public key"),
516 QLatin1String("unusable public key"),
517 QLatin1String("no secret key")}))
518 return QCoreApplication::translate(
519 "Pass", "Encryption failed: recipient GPG key not found or invalid. "
520 "Check that the key ID in .gpg-id is correct and imported.");
521 if (containsAnyCaseInsensitive(err, {QLatin1String("encryption failed")}))
522 return QCoreApplication::translate(
523 "Pass", "Encryption failed. Check that your GPG key is valid.");
524
525 return {};
526}

529auto isGrepHeaderLine(const QString &rawLine, const QString &trimmedLine)
530 -> bool {
531 // ANSI-colored header starts with the blue escape; plain-text fallback:
532 // a non-indented line ending with ':' (pass grep format without color)
533 return rawLine.startsWith(QStringLiteral("\x1B[94m")) ||
534 (!rawLine.startsWith(' ') && !rawLine.startsWith('\t') &&
535 trimmedLine.endsWith(':'));
536}

546auto parseGrepOutput(const QString &rawOut)
547 -> QList<QPair<QString, QStringList>> {
548 static const QRegularExpression ansi(
549 QStringLiteral(R"(\x1B\‍[[0-9;]*[a-zA-Z])"));
550 QList<QPair<QString, QStringList>> results;
551 QString currentEntry;
552 QStringList currentMatches;
553 for (const QString &rawLine : rawOut.split('\n')) {
554 QString line = rawLine;
555 line.remove('\r');
556 line.remove(ansi);
557 line = line.trimmed();
558 const bool isHeader = isGrepHeaderLine(rawLine, line);
559 if (isHeader) {
560 if (!currentEntry.isEmpty() && !currentMatches.isEmpty())
561 results.append({currentEntry, currentMatches});
562 currentEntry = line.endsWith(':') ? line.chopped(1) : line;
563 currentMatches.clear();
564 } else if (!currentEntry.isEmpty()) {
565 if (!line.isEmpty())
566 currentMatches << line;
567 }
568 }
569 if (!currentEntry.isEmpty() && !currentMatches.isEmpty())
570 results.append({currentEntry, currentMatches});
571 return results;
572}

623auto Pass::formatInsertError(const QString &friendly, const QString &err)
624 -> QString {
625 QStringList humanLines;
626 for (const QString &line : err.split('\n')) {
627 QString cleanedLine = line;
628 cleanedLine.remove('\r');
629 if (!cleanedLine.startsWith(QLatin1String("[GNUPG:]")))
630 humanLines.append(cleanedLine);
631 }
632 const QString humanErr = humanLines.join('\n').trimmed();
633 return humanErr.isEmpty() ? friendly : friendly + "\n\n" + humanErr;
634}

724auto Pass::getGpgIdPath(const QString &for_file) -> QString {
725 QString passStore =
726 QDir::fromNativeSeparators(QtPassSettings::getPassStore());
727 QString normalizedFile = QDir::fromNativeSeparators(for_file);
728 QString fullPath = normalizedFile.startsWith(passStore)
729 ? normalizedFile

732 bool found = false;

734 if (QFile(gpgIdDir.absoluteFilePath(".gpg-id")).exists()) {

739 break;
740 }
741 }

743 found ? gpgIdDir.absoluteFilePath(".gpg-id")
744 : QDir(QtPassSettings::getPassStore()).filePath(".gpg-id"));
745
746 return gpgIdPath;

795auto Pass::boundedRandom(quint32 bound) -> quint32 {
796 if (bound < 2) {
797 return 0;
798 }
799
800 quint32 randval;
801 // Rejection-sampling threshold to avoid modulo bias:
802 // In quint32 arithmetic, (1 + ~bound) wraps to (2^32 - bound), so
803 // (1 + ~bound) % bound == 2^32 % bound.
804 // Values randval < rejectionThreshold are rejected; accepted values
805 // produce a uniform distribution when reduced with (randval % bound).
806 const quint32 rejectionThreshold = (1 + ~bound) % bound;
807
808 do {
809 randval = QRandomGenerator::system()->generate();
810 } while (randval < rejectionThreshold);
811
812 return randval % bound;
813}
