63 parts.append(v);
64 }
65 *it = wslenvPrefix + parts.join(':');
66 }

120auto Pass::generatePassword(unsigned int length, const QString &charset)
121 -> QString {
122 QString passwd;
123 if (QtPassSettings::isUsePwgen()) {
124 // --secure goes first as it overrides --no-* otherwise
125 QStringList args;
126 args.append("-1");
127 if (!QtPassSettings::isLessRandom()) {
128 args.append("--secure");
129 }
130 args.append(QtPassSettings::isAvoidCapitals() ? "--no-capitalize"
131 : "--capitalize");
132 args.append(QtPassSettings::isAvoidNumbers() ? "--no-numerals"
133 : "--numerals");
134 if (QtPassSettings::isUseSymbols()) {
135 args.append("--symbols");
136 }
137 args.append(QString::number(length));
138 // executeBlocking returns 0 on success, non-zero on failure
139 if (Executor::executeBlocking(QtPassSettings::getPwgenExecutable(), args,
140 &passwd) == 0) {
141 static const QRegularExpression literalNewLines{"[\\n\\r]"};
142 passwd.remove(literalNewLines);
143 } else {
144 passwd.clear();
145#ifdef QT_DEBUG
146 qDebug() << __FILE__ << ":" << __LINE__ << "\t"
147 << "pwgen fail";
148#endif
149 // Error is already handled by clearing passwd; no need for critical
150 // signal here
151 }
152 } else {
153 // Validate charset - if CUSTOM is selected but chars are empty,
154 // fall back to ALLCHARS to prevent weak passwords (issue #780)
155 QString effectiveCharset = charset;
156 if (effectiveCharset.isEmpty()) {
157 effectiveCharset = QtPassSettings::getPasswordConfiguration()
158 .Characters[PasswordConfiguration::ALLCHARS];
159 }
160 if (effectiveCharset.length() > 0) {
161 passwd = generateRandomPassword(effectiveCharset, length);
162 } else {
163 emit critical(
164 tr("No characters chosen"),
165 tr("Can't generate password, there are no characters to choose from "
166 "set in the configuration!"));
167 }
168 }
169 return passwd;
170}

177bool Pass::gpgSupportsEd25519() {
178 QString out, err;
179 if (Executor::executeBlocking(QtPassSettings::getGpgExecutable(),
180 {"--version"}, &out, &err) != 0) {
181 return false;
182 }
183 QRegularExpression versionRegex(R"(gpg \‍(GnuPG\) (\d+)\.(\d+))");
184 QRegularExpressionMatch match = versionRegex.match(out);
185 if (!match.hasMatch()) {
186 return false;
187 }
188 int major = match.captured(1).toInt();
189 int minor = match.captured(2).toInt();
190 return major > 2 || (major == 2 && minor >= 1);
191}

284QStringList splitCommandCompat(const QString &command) {
285 QStringList result;
286 QString current;
287 bool inSingleQuote = false;
288 bool inDoubleQuote = false;
289 bool escaping = false;
290 for (QChar ch : command) {
291 if (escaping) {
292 current.append(ch);
293 escaping = false;
294 continue;
295 }
296 if (ch == '\\') {
297 escaping = true;
298 continue;
299 }
300 if (ch == '\'' && !inDoubleQuote) {
301 inSingleQuote = !inSingleQuote;
302 continue;
303 }
304 if (ch == '"' && !inSingleQuote) {
305 inDoubleQuote = !inDoubleQuote;
306 continue;
307 }
308 if (ch.isSpace() && !inSingleQuote && !inDoubleQuote) {
309 if (!current.isEmpty()) {
310 result.append(current);
311 current.clear();
312 }
313 continue;
314 }
315 current.append(ch);
316 }
317 if (escaping) {
318 current.append('\\');
319 }
320 if (!current.isEmpty()) {
321 result.append(current);
322 }
323 return result;
324}
325#endif

342auto Pass::resolveGpgconfCommand(const QString &gpgPath)
343 -> ResolvedGpgconfCommand {
344 if (gpgPath.trimmed().isEmpty()) {
345 return {"gpgconf", {}};
346 }
347
348#if QT_VERSION >= QT_VERSION_CHECK(5, 15, 0)
349 QStringList parts = QProcess::splitCommand(gpgPath);
350#else
351 QStringList parts = splitCommandCompat(gpgPath);
352#endif
353
354 if (parts.isEmpty()) {
355 return {"gpgconf", {}};
356 }
357
358 const QString first = parts.first();
359 if (first == "wsl" || first == "wsl.exe") {
360 if (parts.size() >= 2 && parts.at(1).startsWith("sh")) {
361 return {"gpgconf", {}};
362 }
363 if (parts.size() >= 2 &&
364 QFileInfo(parts.last()).fileName().startsWith("gpg")) {
365 QString wslGpgconf = resolveWslGpgconfPath(parts.last());
366 parts.removeLast();
367 parts.append(wslGpgconf);
368 return {parts.first(), parts.mid(1)};
369 }
370 return {"gpgconf", {}};
371 }
372
373 if (!first.contains('/') && !first.contains('\\')) {
374 return {"gpgconf", {}};
375 }
376
377 QString gpgconfPath = findGpgconfInGpgDir(gpgPath);
378 if (!gpgconfPath.isEmpty()) {
379 return {gpgconfPath, {}};
380 }
381
382 return {"gpgconf", {}};
383}

389void Pass::GenerateGPGKeys(QString batch) {
390 // Kill any stale GPG agents that might be holding locks on the key database
391 // This helps avoid "database locked" timeouts during key generation
392 QString gpgPath = QtPassSettings::getGpgExecutable();
393 if (!gpgPath.isEmpty()) {
394 ResolvedGpgconfCommand gpgconf = resolveGpgconfCommand(gpgPath);
395 QStringList killArgs = gpgconf.arguments;
396 killArgs << "--kill";
397 killArgs << "gpg-agent";
398 // Use same environment as key generation to target correct gpg-agent
399 Executor::executeBlocking(env, gpgconf.program, killArgs);
400 }
401
402 executeWrapper(GPG_GENKEYS, gpgPath, {"--gen-key", "--no-tty", "--batch"},
403 std::move(batch));
404}

450auto gpgErrorMessage(const QString &err) -> QString {
451 // Machine-readable status tokens added by --status-fd 2
452 if (err.contains(QStringLiteral("[GNUPG:] KEYEXPIRED")) ||
453 err.contains(QStringLiteral("[GNUPG:] INV_RECP 5 ")))
454 return QCoreApplication::translate("Pass",
455 "Encryption failed: GPG key has expired."
456 " Please renew or replace it.");
457 if (err.contains(QStringLiteral("[GNUPG:] KEYREVOKED")) ||
458 err.contains(QStringLiteral("[GNUPG:] INV_RECP 4 ")))
459 return QCoreApplication::translate(
460 "Pass", "Encryption failed: GPG key has been revoked.");
461 if (err.contains(QStringLiteral("[GNUPG:] NO_PUBKEY")) ||
462 err.contains(QStringLiteral("[GNUPG:] INV_RECP")))
463 return QCoreApplication::translate(
464 "Pass", "Encryption failed: recipient GPG key not found or invalid."
465 " Check that the key ID in .gpg-id is correct and imported.");
466 if (err.contains(QStringLiteral("[GNUPG:] FAILURE")))
467 return QCoreApplication::translate(
468 "Pass", "Encryption failed. Check that your GPG key is valid.");
469
470 // Locale-dependent fallbacks for GPG versions / configurations that don't
471 // emit status tokens
472 const QString lower = err.toLower();
473 if (lower.contains(QLatin1String("key has expired")) ||
474 lower.contains(QLatin1String("key expired")))
475 return QCoreApplication::translate("Pass",
476 "Encryption failed: GPG key has expired."
477 " Please renew or replace it.");
478 if (lower.contains(QLatin1String("key has been revoked")) ||
479 lower.contains(QLatin1String("revoked")))
480 return QCoreApplication::translate(
481 "Pass", "Encryption failed: GPG key has been revoked.");
482 if (lower.contains(QLatin1String("no public key")) ||
483 lower.contains(QLatin1String("unusable public key")) ||
484 lower.contains(QLatin1String("no secret key")))
485 return QCoreApplication::translate(
486 "Pass", "Encryption failed: recipient GPG key not found or invalid."
487 " Check that the key ID in .gpg-id is correct and imported.");
488 if (lower.contains(QLatin1String("encryption failed")))
489 return QCoreApplication::translate(
490 "Pass", "Encryption failed. Check that your GPG key is valid.");
491
492 return {};
493}

505void Pass::finished(int id, int exitCode, const QString &out,
506 const QString &err) {
507 auto pid = static_cast<PROCESS>(id);
508 if (exitCode != 0) {
509 if (pid == PASS_INSERT) {
510 const QString friendly = gpgErrorMessage(err);
511 if (!friendly.isEmpty()) {
512 // Strip machine-readable [GNUPG:] status lines before showing raw
513 // output; they are only useful for detection, not for display.
514 QStringList humanLines;
515 for (QString line : err.split('\n')) {
516 line.remove('\r');
517 if (!line.startsWith(QLatin1String("[GNUPG:]")))
518 humanLines.append(line);
519 }
520 const QString humanErr = humanLines.join('\n').trimmed();
521 emit processErrorExit(exitCode, humanErr.isEmpty()
522 ? friendly
523 : friendly + "\n\n" + humanErr);
524 return;
525 }
526 }
527 emit processErrorExit(exitCode, err);
528 return;
529 }
530 switch (pid) {
531 case GIT_INIT:
532 emit finishedGitInit(out, err);
533 break;
534 case GIT_PULL:
535 emit finishedGitPull(out, err);
536 break;
537 case GIT_PUSH:
538 emit finishedGitPush(out, err);
539 break;
540 case PASS_SHOW:
541 emit finishedShow(out);
542 break;
543 case PASS_OTP_GENERATE:
544 emit finishedOtpGenerate(out);
545 break;
546 case PASS_INSERT:
547 emit finishedInsert(out, err);
548 break;
549 case PASS_REMOVE:
550 emit finishedRemove(out, err);
551 break;
552 case PASS_INIT:
553 emit finishedInit(out, err);
554 break;
555 case PASS_MOVE:
556 emit finishedMove(out, err);
557 break;
558 case PASS_COPY:
559 emit finishedCopy(out, err);
560 break;
561 case GPG_GENKEYS:
562 emit finishedGenerateGPGKeys(out, err);
563 break;
564 default:
565#ifdef QT_DEBUG
566 dbg() << "Unhandled process type" << pid;
567#endif
568 break;
569 }
570}

613auto Pass::getGpgIdPath(const QString &for_file) -> QString {
614 QString passStore =
615 QDir::fromNativeSeparators(QtPassSettings::getPassStore());
616 QString normalizedFile = QDir::fromNativeSeparators(for_file);
617 QString fullPath = normalizedFile.startsWith(passStore)
618 ? normalizedFile
619 : passStore + "/" + normalizedFile;
620 QDir gpgIdDir(QFileInfo(fullPath).absoluteDir());
621 bool found = false;
622 while (gpgIdDir.exists() && gpgIdDir.absolutePath().startsWith(passStore)) {
623 if (QFile(gpgIdDir.absoluteFilePath(".gpg-id")).exists()) {
624 found = true;
625 break;
626 }
627 if (!gpgIdDir.cdUp()) {
628 break;
629 }
630 }
631 QString gpgIdPath(
632 found ? gpgIdDir.absoluteFilePath(".gpg-id")
633 : QDir(QtPassSettings::getPassStore()).filePath(".gpg-id"));
634
635 return gpgIdPath;
636}

684auto Pass::boundedRandom(quint32 bound) -> quint32 {
685 if (bound < 2) {
686 return 0;
687 }
688
689 quint32 randval;
690 // Rejection-sampling threshold to avoid modulo bias:
691 // In quint32 arithmetic, (1 + ~bound) wraps to (2^32 - bound), so
692 // (1 + ~bound) % bound == 2^32 % bound.
693 // Values randval < max_mod_bound are rejected; accepted values produce a
694 // uniform distribution when reduced with (randval % bound).
695 const quint32 max_mod_bound = (1 + ~bound) % bound;
696
697 do {
698 randval = QRandomGenerator::system()->generate();
699 } while (randval < max_mod_bound);
700
701 return randval % bound;
702}

710auto Pass::generateRandomPassword(const QString &charset, unsigned int length)
711 -> QString {
712 if (charset.isEmpty() || length == 0U) {
713 return {};

715 QString out;
