97auto Pass::Generate_b(unsigned int length, const QString &charset) -> QString {
98 QString passwd;
99 if (QtPassSettings::isUsePwgen()) {
100 // --secure goes first as it overrides --no-* otherwise
101 QStringList args;
102 args.append("-1");
103 if (!QtPassSettings::isLessRandom()) {
104 args.append("--secure");
105 }
106 args.append(QtPassSettings::isAvoidCapitals() ? "--no-capitalize"
107 : "--capitalize");
108 args.append(QtPassSettings::isAvoidNumbers() ? "--no-numerals"
109 : "--numerals");
110 if (QtPassSettings::isUseSymbols()) {
111 args.append("--symbols");
112 }
113 args.append(QString::number(length));
114 // executeBlocking returns 0 on success, non-zero on failure
115 if (Executor::executeBlocking(QtPassSettings::getPwgenExecutable(), args,
116 &passwd) == 0) {
117 static const QRegularExpression literalNewLines{"[\\n\\r]"};
118 passwd.remove(literalNewLines);
119 } else {
120 passwd.clear();
121#ifdef QT_DEBUG
122 qDebug() << __FILE__ << ":" << __LINE__ << "\t"
123 << "pwgen fail";
124#endif
125 // Error is already handled by clearing passwd; no need for critical
126 // signal here
127 }
128 } else {
129 // Validate charset - if CUSTOM is selected but chars are empty,
130 // fall back to ALLCHARS to prevent weak passwords (issue #780)
131 QString effectiveCharset = charset;
132 if (effectiveCharset.isEmpty()) {
133 effectiveCharset = QtPassSettings::getPasswordConfiguration()
134 .Characters[PasswordConfiguration::ALLCHARS];
135 }
136 if (effectiveCharset.length() > 0) {
137 passwd = generateRandomPassword(effectiveCharset, length);
138 } else {
139 emit critical(
140 tr("No characters chosen"),
141 tr("Can't generate password, there are no characters to choose from "
142 "set in the configuration!"));
143 }
144 }
145 return passwd;
146}

153bool Pass::gpgSupportsEd25519() {
154 QString out, err;
155 if (Executor::executeBlocking(QtPassSettings::getGpgExecutable(),
156 {"--version"}, &out, &err) != 0) {
157 return false;
158 }
159 QRegularExpression versionRegex(R"(gpg \‍(GnuPG\) (\d+)\.(\d+))");
160 QRegularExpressionMatch match = versionRegex.match(out);
161 if (!match.hasMatch()) {
162 return false;
163 }
164 int major = match.captured(1).toInt();
165 int minor = match.captured(2).toInt();
166 return major > 2 || (major == 2 && minor >= 1);
167}

216auto Pass::listKeys(QStringList keystrings, bool secret) -> QList<UserInfo> {
217 QList<UserInfo> users;
218 QStringList args = {"--no-tty", "--with-colons", "--with-fingerprint"};
219 args.append(secret ? "--list-secret-keys" : "--list-keys");
220
221 for (const QString &keystring : AS_CONST(keystrings)) {
222 if (!keystring.isEmpty()) {
223 args.append(keystring);
224 }
225 }
226 QString p_out;
227 if (Executor::executeBlocking(QtPassSettings::getGpgExecutable(), args,
228 &p_out) != 0) {
229 return users;
230 }
231#if QT_VERSION >= QT_VERSION_CHECK(5, 15, 0)
232 const QStringList keys =
233 p_out.split(Util::newLinesRegex(), Qt::SkipEmptyParts);
234#else
235 const QStringList keys =
236 p_out.split(Util::newLinesRegex(), QString::SkipEmptyParts);
237#endif
238 UserInfo current_user;
239 for (const QString &key : keys) {
240 QStringList props = key.split(':');
241 if (props.size() < 10) {
242 continue;
243 }
244 if (props[0] == (secret ? "sec" : "pub")) {
245 if (!current_user.key_id.isEmpty()) {
246 users.append(current_user);
247 }
248 current_user = UserInfo();
249 current_user.key_id = props[4];
250 current_user.name = props[9].toUtf8();
251 current_user.validity = props[1][0].toLatin1();
252 current_user.created.setSecsSinceEpoch(props[5].toUInt());
253 current_user.expiry.setSecsSinceEpoch(props[6].toUInt());
254 } else if (current_user.name.isEmpty() && props[0] == "uid") {
255 current_user.name = props[9];
256 } else if ((props[0] == "fpr") && props[9].endsWith(current_user.key_id)) {
257 current_user.key_id = props[9];
258 }
259 }
260 if (!current_user.key_id.isEmpty()) {
261 users.append(current_user);
262 }
263 return users;
264}

286void Pass::finished(int id, int exitCode, const QString &out,
287 const QString &err) {
288 auto pid = static_cast<PROCESS>(id);
289 if (exitCode != 0) {
290 emit processErrorExit(exitCode, err);
291 return;
292 }
293 switch (pid) {
294 case GIT_INIT:
295 emit finishedGitInit(out, err);
296 break;
297 case GIT_PULL:
298 emit finishedGitPull(out, err);
299 break;
300 case GIT_PUSH:
301 emit finishedGitPush(out, err);
302 break;
303 case PASS_SHOW:
304 emit finishedShow(out);
305 break;
306 case PASS_OTP_GENERATE:
307 emit finishedOtpGenerate(out);
308 break;
309 case PASS_INSERT:
310 emit finishedInsert(out, err);
311 break;
312 case PASS_REMOVE:
313 emit finishedRemove(out, err);
314 break;
315 case PASS_INIT:
316 emit finishedInit(out, err);
317 break;
318 case PASS_MOVE:
319 emit finishedMove(out, err);
320 break;
321 case PASS_COPY:
322 emit finishedCopy(out, err);
323 break;
324 case GPG_GENKEYS:
325 emit finishedGenerateGPGKeys(out, err);
326 break;
327 default:
328#ifdef QT_DEBUG
329 dbg() << "Unhandled process type" << pid;
330#endif
331 break;
332 }
333}

339void Pass::updateEnv() {
340 // put PASSWORD_STORE_SIGNING_KEY in env
341 QStringList envSigningKey = env.filter("PASSWORD_STORE_SIGNING_KEY=");
342 QString currentSigningKey = QtPassSettings::getPassSigningKey();
343 if (envSigningKey.isEmpty()) {
344 if (!currentSigningKey.isEmpty()) {
345 // dbg()<< "Added
346 // PASSWORD_STORE_SIGNING_KEY with" + currentSigningKey;
347 env.append("PASSWORD_STORE_SIGNING_KEY=" + currentSigningKey);
348 }
349 } else {
350 if (currentSigningKey.isEmpty()) {
351 // dbg() << "Removed
352 // PASSWORD_STORE_SIGNING_KEY";
353 env.removeAll(envSigningKey.first());
354 } else {
355 // dbg()<< "Update
356 // PASSWORD_STORE_SIGNING_KEY with " + currentSigningKey;
357 env.replaceInStrings(envSigningKey.first(),
358 "PASSWORD_STORE_SIGNING_KEY=" + currentSigningKey);
359 }
360 }
361 // put PASSWORD_STORE_DIR in env
362 QStringList store = env.filter("PASSWORD_STORE_DIR=");
363 if (store.isEmpty()) {
364 // dbg()<< "Added
365 // PASSWORD_STORE_DIR";
366 env.append("PASSWORD_STORE_DIR=" + QtPassSettings::getPassStore());
367 } else {
368 // dbg()<< "Update
369 // PASSWORD_STORE_DIR with " + passStore;
370 env.replaceInStrings(store.first(), "PASSWORD_STORE_DIR=" +
371 QtPassSettings::getPassStore());
372 }
373 exec.setEnvironment(env);
374}

381auto Pass::getGpgIdPath(const QString &for_file) -> QString {
382 QString passStore =
383 QDir::fromNativeSeparators(QtPassSettings::getPassStore());
384 QString normalizedFile = QDir::fromNativeSeparators(for_file);
385 QString fullPath = normalizedFile.startsWith(passStore)
386 ? normalizedFile
387 : passStore + "/" + normalizedFile;
388 QDir gpgIdDir(QFileInfo(fullPath).absoluteDir());
389 bool found = false;
390 while (gpgIdDir.exists() && gpgIdDir.absolutePath().startsWith(passStore)) {
391 if (QFile(gpgIdDir.absoluteFilePath(".gpg-id")).exists()) {
392 found = true;
393 break;
394 }
395 if (!gpgIdDir.cdUp()) {
396 break;
397 }
398 }
399 QString gpgIdPath(found ? gpgIdDir.absoluteFilePath(".gpg-id")
400 : QtPassSettings::getPassStore() + ".gpg-id");
401
402 return gpgIdPath;
403}

410auto Pass::getRecipientList(const QString &for_file) -> QStringList {
411 QFile gpgId(getGpgIdPath(for_file));
412 if (!gpgId.open(QIODevice::ReadOnly | QIODevice::Text)) {
413 return {};
414 }
415 QStringList recipients;
416 while (!gpgId.atEnd()) {
417 QString recipient(gpgId.readLine());
418 recipient = recipient.split("#")[0].trimmed();
419 if (!recipient.isEmpty()) {
420 recipients += recipient;
421 }
422 }
423 return recipients;
424}

451auto Pass::boundedRandom(quint32 bound) -> quint32 {
452 if (bound < 2) {
453 return 0;
454 }
455
456 quint32 randval;
457 const quint32 max_mod_bound = (1 + ~bound) % bound;
458
459 do {
460 randval = QRandomGenerator::system()->generate();
461 } while (randval < max_mod_bound);
462
463 return randval % bound;
464}

472auto Pass::generateRandomPassword(const QString &charset, unsigned int length)
473 -> QString {
474 QString out;
475 for (unsigned int i = 0; i < length; ++i) {
476 out.append(charset.at(static_cast<int>(
477 boundedRandom(static_cast<quint32>(charset.length())))));
478 }
479 return out;
480}
