121auto Pass::generatePassword(unsigned int length, const QString &charset)
122 -> QString {
123 QString passwd;
124 if (QtPassSettings::isUsePwgen()) {
125 // --secure goes first as it overrides --no-* otherwise
126 QStringList args;
127 args.append("-1");
128 if (!QtPassSettings::isLessRandom()) {
129 args.append("--secure");
130 }
131 args.append(QtPassSettings::isAvoidCapitals() ? "--no-capitalize"
132 : "--capitalize");
133 args.append(QtPassSettings::isAvoidNumbers() ? "--no-numerals"
134 : "--numerals");
135 if (QtPassSettings::isUseSymbols()) {
136 args.append("--symbols");
137 }
138 args.append(QString::number(length));
139 // executeBlocking returns 0 on success, non-zero on failure
140 if (Executor::executeBlocking(QtPassSettings::getPwgenExecutable(), args,
141 &passwd) == 0) {
142 static const QRegularExpression literalNewLines{"[\\n\\r]"};
143 passwd.remove(literalNewLines);
144 } else {
145 passwd.clear();
146#ifdef QT_DEBUG
147 qDebug() << __FILE__ << ":" << __LINE__ << "\t"
148 << "pwgen fail";
149#endif
150 // Error is already handled by clearing passwd; no need for critical
151 // signal here
152 }
153 } else {
154 // Validate charset - if CUSTOM is selected but chars are empty,
155 // fall back to ALLCHARS to prevent weak passwords (issue #780)
156 QString effectiveCharset = charset;
157 if (effectiveCharset.isEmpty()) {
158 effectiveCharset = QtPassSettings::getPasswordConfiguration()
159 .Characters[PasswordConfiguration::ALLCHARS];
160 }
161 if (effectiveCharset.length() > 0) {
162 passwd = generateRandomPassword(effectiveCharset, length);
163 } else {
164 emit critical(
165 tr("No characters chosen"),
166 tr("Can't generate password, there are no characters to choose from "
167 "set in the configuration!"));
168 }
169 }
170 return passwd;
171}

178bool Pass::gpgSupportsEd25519() {
179 QString out, err;
180 if (Executor::executeBlocking(QtPassSettings::getGpgExecutable(),
181 {"--version"}, &out, &err) != 0) {
182 return false;
183 }
184 QRegularExpression versionRegex(R"(gpg \‍(GnuPG\) (\d+)\.(\d+))");
185 QRegularExpressionMatch match = versionRegex.match(out);
186 if (!match.hasMatch()) {
187 return false;
188 }
189 int major = match.captured(1).toInt();
190 int minor = match.captured(2).toInt();
191 return major > 2 || (major == 2 && minor >= 1);
192}

285QStringList splitCommandCompat(const QString &command) {
286 QStringList result;
287 QString current;
288 bool inSingleQuote = false;
289 bool inDoubleQuote = false;
290 bool escaping = false;
291 for (QChar ch : command) {
292 if (escaping) {
293 current.append(ch);
294 escaping = false;
295 continue;
296 }
297 if (ch == '\\') {
298 escaping = true;
299 continue;
300 }
301 if (ch == '\'' && !inDoubleQuote) {
302 inSingleQuote = !inSingleQuote;
303 continue;
304 }
305 if (ch == '"' && !inSingleQuote) {
306 inDoubleQuote = !inDoubleQuote;
307 continue;
308 }
309 if (ch.isSpace() && !inSingleQuote && !inDoubleQuote) {
310 if (!current.isEmpty()) {
311 result.append(current);
312 current.clear();
313 }
314 continue;
315 }
316 current.append(ch);
317 }
318 if (escaping) {
319 current.append('\\');
320 }
321 if (!current.isEmpty()) {
322 result.append(current);
323 }
324 return result;
325}

343auto Pass::resolveGpgconfCommand(const QString &gpgPath)
344 -> ResolvedGpgconfCommand {
345 if (gpgPath.trimmed().isEmpty()) {
346 return {"gpgconf", {}};
347 }
348
349#if QT_VERSION >= QT_VERSION_CHECK(5, 15, 0)
350 QStringList parts = QProcess::splitCommand(gpgPath);
351#else
352 QStringList parts = splitCommandCompat(gpgPath);
353#endif
354
355 if (parts.isEmpty()) {
356 return {"gpgconf", {}};
357 }
358
359 const QString first = parts.first();
360 if (first == "wsl" || first == "wsl.exe") {
361 if (parts.size() >= 2 && parts.at(1).startsWith("sh")) {
362 return {"gpgconf", {}};
363 }
364 if (parts.size() >= 2 &&
365 QFileInfo(parts.last()).fileName().startsWith("gpg")) {
366 QString wslGpgconf = resolveWslGpgconfPath(parts.last());
367 parts.removeLast();
368 parts.append(wslGpgconf);
369 return {parts.first(), parts.mid(1)};
370 }
371 return {"gpgconf", {}};
372 }
373
374 if (!first.contains('/') && !first.contains('\\')) {
375 return {"gpgconf", {}};
376 }
377
378 QString gpgconfPath = findGpgconfInGpgDir(gpgPath);
379 if (!gpgconfPath.isEmpty()) {
380 return {gpgconfPath, {}};
381 }
382
383 return {"gpgconf", {}};
384}

390void Pass::GenerateGPGKeys(QString batch) {
391 // Kill any stale GPG agents that might be holding locks on the key database
392 // This helps avoid "database locked" timeouts during key generation
393 QString gpgPath = QtPassSettings::getGpgExecutable();
394 if (!gpgPath.isEmpty()) {
395 ResolvedGpgconfCommand gpgconf = resolveGpgconfCommand(gpgPath);
396 QStringList killArgs = gpgconf.arguments;
397 killArgs << "--kill";
398 killArgs << "gpg-agent";
399 // Use same environment as key generation to target correct gpg-agent
400 Executor::executeBlocking(env, gpgconf.program, killArgs);
401 }
402
403 executeWrapper(GPG_GENKEYS, gpgPath, {"--gen-key", "--no-tty", "--batch"},
404 std::move(batch));
405}

451auto gpgErrorMessage(const QString &err) -> QString {
452 // Machine-readable status tokens added by --status-fd 2
453 if (err.contains(QStringLiteral("[GNUPG:] KEYEXPIRED")) ||
454 err.contains(QStringLiteral("[GNUPG:] INV_RECP 5 ")))
455 return QCoreApplication::translate("Pass",
456 "Encryption failed: GPG key has expired."
457 " Please renew or replace it.");
458 if (err.contains(QStringLiteral("[GNUPG:] KEYREVOKED")) ||
459 err.contains(QStringLiteral("[GNUPG:] INV_RECP 4 ")))
460 return QCoreApplication::translate(
461 "Pass", "Encryption failed: GPG key has been revoked.");
462 if (err.contains(QStringLiteral("[GNUPG:] NO_PUBKEY")) ||
463 err.contains(QStringLiteral("[GNUPG:] INV_RECP")))
464 return QCoreApplication::translate(
465 "Pass", "Encryption failed: recipient GPG key not found or invalid."
466 " Check that the key ID in .gpg-id is correct and imported.");
467 if (err.contains(QStringLiteral("[GNUPG:] FAILURE")))
468 return QCoreApplication::translate(
469 "Pass", "Encryption failed. Check that your GPG key is valid.");
470
471 // Locale-dependent fallbacks for GPG versions / configurations that don't
472 // emit status tokens
473 const QString lower = err.toLower();
474 if (lower.contains(QLatin1String("key has expired")) ||
475 lower.contains(QLatin1String("key expired")))
476 return QCoreApplication::translate("Pass",
477 "Encryption failed: GPG key has expired."
478 " Please renew or replace it.");
479 if (lower.contains(QLatin1String("key has been revoked")) ||
480 lower.contains(QLatin1String("revoked")))
481 return QCoreApplication::translate(
482 "Pass", "Encryption failed: GPG key has been revoked.");
483 if (lower.contains(QLatin1String("no public key")) ||
484 lower.contains(QLatin1String("unusable public key")) ||
485 lower.contains(QLatin1String("no secret key")))
486 return QCoreApplication::translate(
487 "Pass", "Encryption failed: recipient GPG key not found or invalid."
488 " Check that the key ID in .gpg-id is correct and imported.");
489 if (lower.contains(QLatin1String("encryption failed")))
490 return QCoreApplication::translate(
491 "Pass", "Encryption failed. Check that your GPG key is valid.");
492
493 return {};
494}

503auto parseGrepOutput(const QString &rawOut)
504 -> QList<QPair<QString, QStringList>> {
505 static const QRegularExpression ansi(
506 QStringLiteral(R"(\x1B\‍[[0-9;]*[a-zA-Z])"));
507 QList<QPair<QString, QStringList>> results;
508 QString currentEntry;
509 QStringList currentMatches;
510 for (const QString &rawLine : rawOut.split('\n')) {
511 QString line = rawLine;
512 line.remove('\r');
513 line.remove(ansi);
514 line = line.trimmed();
515 // ANSI-colored header starts with the blue escape; plain-text fallback:
516 // a non-indented line ending with ':' (pass grep format without color)
517 bool isHeader = rawLine.startsWith(QStringLiteral("\x1B[94m")) ||
518 (!rawLine.startsWith(' ') && !rawLine.startsWith('\t') &&
519 line.endsWith(':') && !line.isEmpty());
520 if (isHeader) {
521 if (!currentEntry.isEmpty() && !currentMatches.isEmpty())
522 results.append({currentEntry, currentMatches});
523 currentEntry = line.endsWith(':') ? line.chopped(1) : line;
524 currentMatches.clear();
525 } else if (!currentEntry.isEmpty()) {
526 if (!line.isEmpty())
527 currentMatches << line;
528 }
529 }
530 if (!currentEntry.isEmpty() && !currentMatches.isEmpty())
531 results.append({currentEntry, currentMatches});
532 return results;
533}

545void Pass::finished(int id, int exitCode, const QString &out,
546 const QString &err) {
547 auto pid = static_cast<PROCESS>(id);
548 if (exitCode != 0) {
549 if (pid == PASS_GREP) {
550 // exit code 1 means no matches (standard grep behaviour); treat as empty
551 // result set rather than an error
552 if (exitCode == 1) {
553 emit finishedGrep({});
554 } else {
555 emit processErrorExit(exitCode, err);
556 emit finishedGrep({});
557 }
558 return;
559 }
560 if (pid == PASS_INSERT) {
561 const QString friendly = gpgErrorMessage(err);
562 if (!friendly.isEmpty()) {
563 // Strip machine-readable [GNUPG:] status lines before showing raw
564 // output; they are only useful for detection, not for display.
565 QStringList humanLines;
566 for (QString line : err.split('\n')) {
567 line.remove('\r');
568 if (!line.startsWith(QLatin1String("[GNUPG:]")))
569 humanLines.append(line);
570 }
571 const QString humanErr = humanLines.join('\n').trimmed();
572 emit processErrorExit(exitCode, humanErr.isEmpty()
573 ? friendly
574 : friendly + "\n\n" + humanErr);
575 return;
576 }
577 }
578 emit processErrorExit(exitCode, err);
579 return;
580 }
581 switch (pid) {
582 case GIT_INIT:
583 emit finishedGitInit(out, err);
584 break;
585 case GIT_PULL:
586 emit finishedGitPull(out, err);
587 break;
588 case GIT_PUSH:
589 emit finishedGitPush(out, err);
590 break;
591 case PASS_SHOW:
592 emit finishedShow(out);
593 break;
594 case PASS_OTP_GENERATE:
595 emit finishedOtpGenerate(out);
596 break;
597 case PASS_INSERT:
598 emit finishedInsert(out, err);
599 break;
600 case PASS_REMOVE:
601 emit finishedRemove(out, err);
602 break;
603 case PASS_INIT:
604 emit finishedInit(out, err);
605 break;
606 case PASS_MOVE:
607 emit finishedMove(out, err);
608 break;
609 case PASS_COPY:
610 emit finishedCopy(out, err);
611 break;
612 case GPG_GENKEYS:
613 emit finishedGenerateGPGKeys(out, err);
614 break;
615 case PASS_GREP:
616 emit finishedGrep(parseGrepOutput(out));
617 break;
618 default:
619#ifdef QT_DEBUG
620 dbg() << "Unhandled process type" << pid;
621#endif
622 break;
623 }
624}

667auto Pass::getGpgIdPath(const QString &for_file) -> QString {
668 QString passStore =
669 QDir::fromNativeSeparators(QtPassSettings::getPassStore());
670 QString normalizedFile = QDir::fromNativeSeparators(for_file);
671 QString fullPath = normalizedFile.startsWith(passStore)
672 ? normalizedFile
673 : passStore + "/" + normalizedFile;
674 QDir gpgIdDir(QFileInfo(fullPath).absoluteDir());
675 bool found = false;
676 while (gpgIdDir.exists() && gpgIdDir.absolutePath().startsWith(passStore)) {
677 if (QFile(gpgIdDir.absoluteFilePath(".gpg-id")).exists()) {
678 found = true;
679 break;
680 }
681 if (!gpgIdDir.cdUp()) {
682 break;
683 }
684 }
685 QString gpgIdPath(
686 found ? gpgIdDir.absoluteFilePath(".gpg-id")
687 : QDir(QtPassSettings::getPassStore()).filePath(".gpg-id"));
688
689 return gpgIdPath;

738auto Pass::boundedRandom(quint32 bound) -> quint32 {
739 if (bound < 2) {
740 return 0;
741 }
742
743 quint32 randval;
744 // Rejection-sampling threshold to avoid modulo bias:
745 // In quint32 arithmetic, (1 + ~bound) wraps to (2^32 - bound), so
746 // (1 + ~bound) % bound == 2^32 % bound.
747 // Values randval < max_mod_bound are rejected; accepted values produce a
748 // uniform distribution when reduced with (randval % bound).
749 const quint32 max_mod_bound = (1 + ~bound) % bound;
750
751 do {
752 randval = QRandomGenerator::system()->generate();
753 } while (randval < max_mod_bound);
754
755 return randval % bound;
756}
