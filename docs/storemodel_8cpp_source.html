60auto StoreModel::showThis(const QModelIndex index) const -> bool {
61 bool retVal = false;
62 if (fs == nullptr) {
63 return retVal;
64 }
65 // Gives you the info for number of childs with a parent
66 if (sourceModel()->rowCount(index) > 0) {
67 for (int nChild = 0; nChild < sourceModel()->rowCount(index); ++nChild) {
68 QModelIndex childIndex = sourceModel()->index(nChild, 0, index);
69 if (!childIndex.isValid()) {
70 break;
71 }
72 retVal = showThis(childIndex);
73 if (retVal) {
74 break;
75 }
76 }
77 } else {
78 QModelIndex useIndex = sourceModel()->index(index.row(), 0, index.parent());
79 QString path = fs->filePath(useIndex);
80 path = QDir(store).relativeFilePath(path);
81 if (path.startsWith(".git")) {
82 return false;
83 }
84 path.replace(Util::endsWithGpg(), "");
85 retVal = path.contains(filterRegularExpression());
86 }
87 return retVal;
88}

206auto StoreModel::canDropMimeData(const QMimeData *data, Qt::DropAction action,
207 int row, int column,
208 const QModelIndex &parent) const -> bool {
209#ifdef QT_DEBUG
210 qDebug() << action << row;
211#else
212 Q_UNUSED(action)
213 Q_UNUSED(row)
214#endif
215
216 QModelIndex useIndex =
217 this->index(parent.row(), parent.column(), parent.parent());
218 QByteArray encodedData =
219 data->data("application/vnd+qtpass.dragAndDropInfoPasswordStore");
220 QDataStream stream(&encodedData, QIODevice::ReadOnly);
221 dragAndDropInfoPasswordStore info;
222 stream >> info;
223 if (!data->hasFormat("application/vnd+qtpass.dragAndDropInfoPasswordStore")) {
224 return false;
225 }
226
227 if (column > 0) {
228 return false;
229 }
230
231 // you can drop a folder on a folder
232 if (fs->fileInfo(mapToSource(useIndex)).isDir() && info.isDir) {
233 return true;
234 }
235 // you can drop a file on a folder
236 if (fs->fileInfo(mapToSource(useIndex)).isDir() && info.isFile) {
237 return true;
238 }
239 // you can drop a file on a file
240 if (fs->fileInfo(mapToSource(useIndex)).isFile() && info.isFile) {
241 return true;
242 }
243
244 return false;
245}

256auto StoreModel::dropMimeData(const QMimeData *data, Qt::DropAction action,
257 int row, int column, const QModelIndex &parent)
258 -> bool {
259 if (!canDropMimeData(data, action, row, column, parent)) {
260 return false;
261 }
262
263 if (action == Qt::IgnoreAction) {
264 return true;
265 }
266 QByteArray encodedData =
267 data->data("application/vnd+qtpass.dragAndDropInfoPasswordStore");
268
269 QDataStream stream(&encodedData, QIODevice::ReadOnly);
270 dragAndDropInfoPasswordStore info;
271 stream >> info;
272 QModelIndex destIndex =
273 this->index(parent.row(), parent.column(), parent.parent());
274 QFileInfo destFileinfo = fs->fileInfo(mapToSource(destIndex));
275 QFileInfo srcFileInfo = QFileInfo(info.path);
276 QString cleanedSrc = QDir::cleanPath(srcFileInfo.absoluteFilePath());
277 QString cleanedDest = QDir::cleanPath(destFileinfo.absoluteFilePath());
278 if (info.isDir) {
279 // dropped dir onto dir
280 if (destFileinfo.isDir()) {
281 QDir destDir = QDir(cleanedDest).filePath(srcFileInfo.fileName());
282 QString cleanedDestDir = QDir::cleanPath(destDir.absolutePath());
283 if (action == Qt::MoveAction) {
284 QtPassSettings::getPass()->Move(cleanedSrc, cleanedDestDir);
285 } else if (action == Qt::CopyAction) {
286 QtPassSettings::getPass()->Copy(cleanedSrc, cleanedDestDir);
287 }
288 }
289 } else if (info.isFile) {
290 // dropped file onto a directory
291 if (destFileinfo.isDir()) {
292 if (action == Qt::MoveAction) {
293 QtPassSettings::getPass()->Move(cleanedSrc, cleanedDest);
294 } else if (action == Qt::CopyAction) {
295 QtPassSettings::getPass()->Copy(cleanedSrc, cleanedDest);
296 }
297 } else if (destFileinfo.isFile()) {
298 // dropped file onto a file
299 int answer = QMessageBox::question(
300 nullptr, tr("force overwrite?"),
301 tr("overwrite %1 with %2?").arg(cleanedDest, cleanedSrc),
302 QMessageBox::Yes | QMessageBox::No);
303 bool force = answer == QMessageBox::Yes;
304 if (action == Qt::MoveAction) {
305 QtPassSettings::getPass()->Move(cleanedSrc, cleanedDest, force);
306 } else if (action == Qt::CopyAction) {
307 QtPassSettings::getPass()->Copy(cleanedSrc, cleanedDest, force);
308 }
309 }
310 }
311 return true;
312}

320auto StoreModel::lessThan(const QModelIndex &source_left,
321 const QModelIndex &source_right) const -> bool {
322/* matches logic in QFileSystemModelSorter::compareNodes() */
323#ifndef Q_OS_MAC
324 if (fs && (source_left.column() == 0 || source_left.column() == 1)) {
325 bool leftD = fs->isDir(source_left);
326 bool rightD = fs->isDir(source_right);
327
328 if (leftD ^ rightD) {
329 return leftD;
330 }
331 }
332#endif
333
334 return QSortFilterProxyModel::lessThan(source_left, source_right);
335}
