60auto StoreModel::showThis(const QModelIndex &index) const -> bool {
61 bool retVal = false;
62 if (fs == nullptr) {
63 return retVal;
64 }
65 // Gives you the info for number of childs with a parent
66 if (sourceModel()->rowCount(index) > 0) {
67 for (int nChild = 0; nChild < sourceModel()->rowCount(index); ++nChild) {
68 QModelIndex childIndex = sourceModel()->index(nChild, 0, index);
69 if (!childIndex.isValid()) {
70 break;
71 }
72 retVal = showThis(childIndex);
73 if (retVal) {
74 break;
75 }
76 }
77 } else {
78 QModelIndex useIndex = sourceModel()->index(index.row(), 0, index.parent());
79 QString path = fs->filePath(useIndex);
80 path = QDir(store).relativeFilePath(path);
81 if (path.startsWith(".git")) {
82 return false;
83 }
84 path.replace(Util::endsWithGpg(), "");
85 retVal = path.contains(filterRegularExpression());
86 }
87 return retVal;
88}

216auto StoreModel::canDropMimeData(const QMimeData *data, Qt::DropAction action,
217 int row, int column,
218 const QModelIndex &parent) const -> bool {
219#ifdef QT_DEBUG
220 qDebug() << action << row;
221#else
222 Q_UNUSED(action)
223 Q_UNUSED(row)
224#endif
225
226 if (data == nullptr ||
227 !data->hasFormat("application/vnd+qtpass.dragAndDropInfoPasswordStore")) {
228 return false;
229 }
230
231 QByteArray encodedData =
232 data->data("application/vnd+qtpass.dragAndDropInfoPasswordStore");
233 if (encodedData.isEmpty()) {
234 return false;
235 }
236 QDataStream stream(&encodedData, QIODevice::ReadOnly);
237 dragAndDropInfoPasswordStore info;
238 stream >> info;
239 if (stream.status() != QDataStream::Ok) {
240 return false;
241 }
242
243 QModelIndex useIndex =
244 this->index(parent.row(), parent.column(), parent.parent());
245
246 if (column > 0) {
247 return false;
248 }
249
250 // you can drop a folder on a folder
251 if (fs->fileInfo(mapToSource(useIndex)).isDir() && info.isDir) {
252 return true;
253 }
254 // you can drop a file on a folder
255 if (fs->fileInfo(mapToSource(useIndex)).isDir() && info.isFile) {
256 return true;
257 }
258 // you can drop a file on a file
259 if (fs->fileInfo(mapToSource(useIndex)).isFile() && info.isFile) {
260 return true;
261 }
262
263 return false;
264}

275auto StoreModel::dropMimeData(const QMimeData *data, Qt::DropAction action,
276 int row, int column, const QModelIndex &parent)
277 -> bool {
278 if (!canDropMimeData(data, action, row, column, parent)) {
279 return false;
280 }
281
282 if (action == Qt::IgnoreAction) {
283 return true;
284 }
285 QByteArray encodedData =
286 data->data("application/vnd+qtpass.dragAndDropInfoPasswordStore");
287 if (encodedData.isEmpty()) {
288 return false;
289 }
290 QDataStream stream(&encodedData, QIODevice::ReadOnly);
291 dragAndDropInfoPasswordStore info;
292 stream >> info;
293 if (stream.status() != QDataStream::Ok) {
294 return false;
295 }
296 QModelIndex destIndex =
297 this->index(parent.row(), parent.column(), parent.parent());
298 QFileInfo destFileinfo = fs->fileInfo(mapToSource(destIndex));
299 QFileInfo srcFileInfo = QFileInfo(info.path);
300 QString cleanedSrc = QDir::cleanPath(srcFileInfo.absoluteFilePath());
301 QString cleanedDest = QDir::cleanPath(destFileinfo.absoluteFilePath());
302 if (info.isDir) {
303 // dropped dir onto dir
304 if (destFileinfo.isDir()) {
305 QDir destDir = QDir(cleanedDest).filePath(srcFileInfo.fileName());
306 QString cleanedDestDir = QDir::cleanPath(destDir.absolutePath());
307 if (action == Qt::MoveAction) {
308 QtPassSettings::getPass()->Move(cleanedSrc, cleanedDestDir);
309 } else if (action == Qt::CopyAction) {
310 QtPassSettings::getPass()->Copy(cleanedSrc, cleanedDestDir);
311 }
312 }
313 } else if (info.isFile) {
314 // dropped file onto a directory
315 if (destFileinfo.isDir()) {
316 if (action == Qt::MoveAction) {
317 QtPassSettings::getPass()->Move(cleanedSrc, cleanedDest);
318 } else if (action == Qt::CopyAction) {
319 QtPassSettings::getPass()->Copy(cleanedSrc, cleanedDest);
320 }
321 } else if (destFileinfo.isFile()) {
322 // dropped file onto a file
323 int answer = QMessageBox::question(
324 nullptr, tr("force overwrite?"),
325 tr("overwrite %1 with %2?").arg(cleanedDest, cleanedSrc),
326 QMessageBox::Yes | QMessageBox::No);
327 bool force = answer == QMessageBox::Yes;
328 if (action == Qt::MoveAction) {
329 QtPassSettings::getPass()->Move(cleanedSrc, cleanedDest, force);
330 } else if (action == Qt::CopyAction) {
331 QtPassSettings::getPass()->Copy(cleanedSrc, cleanedDest, force);
332 }
333 }
334 }
335 return true;
336}

344auto StoreModel::lessThan(const QModelIndex &source_left,
345 const QModelIndex &source_right) const -> bool {
346/* matches logic in QFileSystemModelSorter::compareNodes() */
347#ifndef Q_OS_MAC
348 if (fs && (source_left.column() == 0 || source_left.column() == 1)) {
349 bool leftD = fs->isDir(source_left);
350 bool rightD = fs->isDir(source_right);
351
352 if (leftD ^ rightD) {
353 return leftD;
354 }
355 }
356#endif
357
358 return QSortFilterProxyModel::lessThan(source_left, source_right);
