73auto StoreModel::showThis(const QModelIndex &index) const -> bool {
74 bool retVal = false;
75 if (fs == nullptr) {
76 return retVal;
77 }
78 // Gives you the info for number of children with a parent
79 if (sourceModel()->rowCount(index) > 0) {
80 for (int nChild = 0; nChild < sourceModel()->rowCount(index); ++nChild) {
81 QModelIndex childIndex = sourceModel()->index(nChild, 0, index);
82 if (!childIndex.isValid()) {
83 break;
84 }
85 retVal = showThis(childIndex);
86 if (retVal) {
87 break;
88 }
89 }
90 } else {
91 QModelIndex useIndex = sourceModel()->index(index.row(), 0, index.parent());
92 QString path = fs->filePath(useIndex);
93 path = QDir(store).relativeFilePath(path);
94 if (path.startsWith(".git")) {
95 return false;
96 }
97 path.replace(Util::endsWithGpg(), "");
98 retVal = path.contains(filterRegularExpression());
99 }
100 return retVal;
101}

233auto StoreModel::canDropMimeData(const QMimeData *data, Qt::DropAction action,
234 int row, int column,
235 const QModelIndex &parent) const -> bool {
236#ifdef QT_DEBUG
237 qDebug() << action << row;
238#else
239 Q_UNUSED(action)
240 Q_UNUSED(row)
241#endif
242
243 if (data == nullptr ||
244 !data->hasFormat("application/vnd+qtpass.dragAndDropInfoPasswordStore")) {
245 return false;
246 }
247
248 QByteArray encodedData =
249 data->data("application/vnd+qtpass.dragAndDropInfoPasswordStore");
250 if (encodedData.isEmpty()) {
251 return false;
252 }
253 QDataStream stream(&encodedData, QIODevice::ReadOnly);
254 dragAndDropInfoPasswordStore info;
255 stream >> info;
256 if (stream.status() != QDataStream::Ok) {
257 return false;
258 }
259
260 QModelIndex useIndex =
261 this->index(parent.row(), parent.column(), parent.parent());
262
263 if (column > 0) {
264 return false;
265 }
266
267 using IK = dragAndDropInfoPasswordStore::ItemKind;
268 // you can drop a folder on a folder
269 if (fs->fileInfo(mapToSource(useIndex)).isDir() &&
270 info.kind == IK::Directory) {
271 return true;
272 }
273 // you can drop a file on a folder
274 if (fs->fileInfo(mapToSource(useIndex)).isDir() && info.kind == IK::File) {
275 return true;
276 }
277 // you can drop a file on a file
278 if (fs->fileInfo(mapToSource(useIndex)).isFile() && info.kind == IK::File) {
279 return true;
280 }
281
282 return false;
283}

295 int row, int column, const QModelIndex &parent)
296 -> bool {

336auto StoreModel::executeDropAction(const dragAndDropInfoPasswordStore &info,
337 Qt::DropAction action,
338 const QModelIndex &parent) -> bool {
339 QModelIndex destIndex =
340 this->index(parent.row(), parent.column(), parent.parent());
341 QFileInfo destFileinfo = fs->fileInfo(mapToSource(destIndex));
342 QFileInfo srcFileInfo = QFileInfo(info.path);
343
344 QString cleanedSrc = QDir::cleanPath(srcFileInfo.absoluteFilePath());
345 QString cleanedDest = QDir::cleanPath(destFileinfo.absoluteFilePath());
346
347 switch (info.kind) {
348 case dragAndDropInfoPasswordStore::ItemKind::Directory:
349 return handleDirDrop(cleanedSrc, destFileinfo, srcFileInfo, action);
350 case dragAndDropInfoPasswordStore::ItemKind::File:
351 return handleFileDrop(cleanedSrc, cleanedDest, destFileinfo, action);
352 default:
353 qWarning() << "executeDropAction: unexpected ItemKind, ignoring drop";
354 return false;
355 }
356}

358auto StoreModel::handleDirDrop(const QString &cleanedSrc,
359 const QFileInfo &destFileinfo,
360 const QFileInfo &srcFileInfo,
361 Qt::DropAction action) -> bool {
362 if (!destFileinfo.isDir()) {
363 return false;
364 }
365
366 QDir destDir = QDir(QDir::cleanPath(destFileinfo.absoluteFilePath()))
367 .filePath(srcFileInfo.fileName());
368 QString cleanedDestDir = QDir::cleanPath(destDir.absolutePath());
369
370 if (action == Qt::MoveAction) {
371 QtPassSettings::getPass()->Move(cleanedSrc, cleanedDestDir);
372 } else if (action == Qt::CopyAction) {
373 QtPassSettings::getPass()->Copy(cleanedSrc, cleanedDestDir);
374 }
375 return true;
