63void tst_gpgkeystate::parseSecretKeys() {
64 QFETCH(QString, input);
65 QFETCH(int, expectedCount);
66 QFETCH(bool, expectHaveSecret);
67
68 const bool includeSecretKeys = true;
69 QList<UserInfo> result = parseGpgColonOutput(input, includeSecretKeys);
70
71 QVERIFY2(result.size() == expectedCount,
72 qPrintable(QString("Expected %1 keys, got %2")
73 .arg(expectedCount)
74 .arg(result.size())));

76 if (expectedCount > 0) {
77 QVERIFY(!result.isEmpty());
78 for (int i = 0; i < result.size(); ++i) {

140void tst_gpgkeystate::parseKeyRollover() {
141 QFETCH(QString, input);
142 QFETCH(int, expectedCount);
143
144 const bool includeSecretKeys = false;
145 QList<UserInfo> result = parseGpgColonOutput(input, includeSecretKeys);
146 QVERIFY2(result.size() == expectedCount,
147 qPrintable(QString("Expected %1 keys, got %2")
148 .arg(expectedCount)
149 .arg(result.size())));
150
151 auto containsKeyId = [&](const QString &keyId) {
152 for (const UserInfo &user : result) {
153 if (user.key_id == keyId) {
154 return true;
155 }
156 }
157 return false;
158 };
159
160 QVERIFY2(containsKeyId("AAA111"), "Expected AAA111 key to be parsed");
161 QVERIFY2(containsKeyId("BBB222"), "Expected BBB222 key to be parsed");
162 QVERIFY2(containsKeyId("CCC333"), "Expected CCC333 key to be parsed");
163}
