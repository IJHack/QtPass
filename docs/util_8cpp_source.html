31void Util::initialiseEnvironment() {
32 if (!_envInitialised) {
33 _env = QProcessEnvironment::systemEnvironment();
34#ifdef __APPLE__
35 QString path = _env.value("PATH");
36 if (!path.contains("/usr/local/MacGPG2/bin") &&
37 QDir("/usr/local/MacGPG2/bin").exists())
38 path += ":/usr/local/MacGPG2/bin";
39 if (!path.contains("/usr/local/bin"))
40 path += ":/usr/local/bin";
41 _env.insert("PATH", path);
42#endif
43#ifdef Q_OS_WIN
44 QString path = _env.value("PATH");
45 if (!path.contains("C:\\Program Files\\WinGPG\\x86") &&
46 QDir("C:\\Program Files\\WinGPG\\x86").exists())
47 path += ";C:\\Program Files\\WinGPG\\x86";
48 if (!path.contains("C:\\Program Files\\GnuPG\\bin") &&
49 QDir("C:\\Program Files\\GnuPG\\bin").exists())
50 path += ";C:\\Program Files\\GnuPG\\bin";
51 _env.insert("PATH", path);
52#endif
53#ifdef QT_DEBUG
54 dbg() << _env.value("PATH");
55#endif
56 _envInitialised = true;
57 }
58}

86auto Util::findBinaryInPath(QString binary) -> QString {
87 initialiseEnvironment();
88
89 QString ret;
90
91 const QString binaryWithSep = QDir::separator() + binary;
92
93 if (_env.contains("PATH")) {
94 QString path = _env.value("PATH");
95 const QChar delimiter = QDir::separator() == '\\' ? ';' : ':';
96 QStringList entries = path.split(delimiter);
97
98 for (const QString &entryConst : entries) {
99 QString fullPath = entryConst + binaryWithSep;
100 QFileInfo qfi(fullPath);
101#ifdef Q_OS_WIN
102 if (!qfi.exists()) {
103 QString fullPathExe = fullPath + ".exe";
104 qfi = QFileInfo(fullPathExe);
105 }
106#endif
107 if (!qfi.exists()) {
108 continue;
109 }
110 if (!qfi.isExecutable()) {
111 continue;
112 }
113
114 ret = qfi.absoluteFilePath();
115 break;
116 }
117 }
118#ifdef Q_OS_WIN
119 if (ret.isEmpty()) {
120 // Validate binary name before attempting WSL fallback: require a non-empty,
121 // whitespace-free program name to avoid confusing WSL invocations.
122 const bool hasWhitespace =
123 binary.contains(QRegularExpression(QStringLiteral("\\s")));
124 if (!binary.isEmpty() && !hasWhitespace) {
125 QString wslCommand = QStringLiteral("wsl ") + binary;
126#ifdef QT_DEBUG
127 dbg() << "Util::findBinaryInPath(): falling back to WSL for binary"
128 << binary;
129#endif
130 QString out, err;
131 if (Executor::executeBlocking(wslCommand, {"--version"}, &out, &err) ==
132 0 &&
133 !out.isEmpty() && err.isEmpty()) {
134#ifdef QT_DEBUG
135 dbg() << "Util::findBinaryInPath(): using WSL binary" << wslCommand;
136#endif
137 ret = wslCommand;
138 }
139 }
140 }
141#endif
142
143 return ret;
144}

146auto Util::configIsValid() -> bool {
147 const QString configFilePath =
148 QDir(QtPassSettings::getPassStore()).filePath(".gpg-id");
149 if (!QFile(configFilePath).exists()) {
150 return false;
151 }
152
153 const QString executable = QtPassSettings::isUsePass()
154 ? QtPassSettings::getPassExecutable()
155 : QtPassSettings::getGpgExecutable();
156
157 if (executable.startsWith(QStringLiteral("wsl "))) {
158 QString out;
159 QString err;
160 if (Executor::executeBlocking(QStringLiteral("wsl"),
161 {QStringLiteral("--version")}, &out,
162 &err) == 0 &&
163 !out.isEmpty() && err.isEmpty()) {
164 return true;
165 }
166 }
167 return QFile(executable).exists();
168}

170auto Util::getDir(const QModelIndex &index, bool forPass,
171 const QFileSystemModel &model, const StoreModel &storeModel)
172 -> QString {
173 QString abspath =
174 QDir(QtPassSettings::getPassStore()).absolutePath() + QDir::separator();
175 if (!index.isValid()) {
176 return forPass ? "" : abspath;
177 }
178 QFileInfo info = model.fileInfo(storeModel.mapToSource(index));
179 QString filePath =
180 (info.isFile() ? info.absolutePath() : info.absoluteFilePath());
181 if (forPass) {
182 filePath = QDir(abspath).relativeFilePath(filePath);
183 }
184 filePath += QDir::separator();
185 return filePath;
186}
