54void tst_settings::initTestCase() {
55 // Check for portable mode (qtpass.ini in app directory)
56 // Only backup/restore settings file in portable mode
57 // On non-portable (registry on Windows), we cannot safely backup
58 QString portable_ini =
59 QCoreApplication::applicationDirPath() + QDir::separator() + "qtpass.ini";
60 m_isPortableMode = QFile::exists(portable_ini);
61
62 if (m_isPortableMode) {
63 QtPassSettings::getInstance()->sync();
64 QString settingsFile = QtPassSettings::getInstance()->fileName();
65 m_settingsBackupPath = settingsFile + ".bak";
66 QFile::remove(m_settingsBackupPath);
67 QVERIFY(QFile::copy(settingsFile, m_settingsBackupPath));
68 } else {
69 m_settingsBackupPath.clear();
70 // On non-portable mode, warn but continue (tests may modify registry)
71 qWarning()
72 << "Non-portable mode detected. Tests may modify registry settings.";
73 }
74}
75
76void tst_settings::cleanupTestCase() {
77 // Restore original settings after all tests
78 // This ensures make check doesn't change user's live config
79 if (m_isPortableMode && !m_settingsBackupPath.isEmpty()) {
80 QString settingsFile = QtPassSettings::getInstance()->fileName();
81 QFile::remove(settingsFile);
82 QVERIFY2(QFile::remove(settingsFile) || !QFile::exists(settingsFile),
83 "Failed to remove current settings file before restore");
84 QVERIFY2(QFile::copy(m_settingsBackupPath, settingsFile),
85 "Failed to restore settings file from backup");
86 QFile::remove(m_settingsBackupPath);
87 }
88}

227void tst_settings::boolRoundTrip() {
228 QFETCH(QString, setting);
229 QFETCH(bool, testValue);
230
231 for (const auto &s : boolSettings) {
232 if (setting == s.name) {
233 s.setter(testValue);
234 QVERIFY2(s.getter(!testValue) == testValue,
235 qPrintable(QString("%1 should be %2, got %3")
236 .arg(setting)
237 .arg(testValue ? "true" : "false")
238 .arg(s.getter(!testValue) ? "true" : "false")));
239 return;
240 }
241 }
242 QFAIL(qPrintable(QString("Unknown setting: %1").arg(setting)));
243}

372void tst_settings::autoDetectGit() {
373 QTemporaryDir tempDir;
374 QtPassSettings::setPassStore(tempDir.path());
375
376 QDir gitDir(tempDir.path());
377 QVERIFY(gitDir.mkdir(".git"));
378 QtPassSettings::getInstance()->sync();
379
380 QtPassSettings::getInstance()->remove("useGit");
381 QtPassSettings::getInstance()->sync();
382 QVERIFY2(QtPassSettings::isUseGit(true),
383 "Should auto-detect .git and return true when default is true");
384
385 QtPassSettings::getInstance()->remove("useGit");
386 QtPassSettings::getInstance()->sync();
387 QVERIFY2(!QtPassSettings::isUseGit(false),
388 "Should return false when default is false, even if .git exists");
389
390 QVERIFY(gitDir.rmdir(".git"));
391 QtPassSettings::getInstance()->sync();
392
393 QtPassSettings::getInstance()->remove("useGit");
394 QtPassSettings::getInstance()->sync();
395 QVERIFY2(QtPassSettings::isUseGit(true),
396 "Should return true default when .git not present");
397
398 QtPassSettings::getInstance()->remove("useGit");
399 QtPassSettings::getInstance()->sync();
400 QVERIFY2(!QtPassSettings::isUseGit(false),
401 "Should return false default when .git not present");
402}
