59void tst_settings::initTestCase() {
60 // Check for portable mode (qtpass.ini in app directory)
61 // Only backup/restore settings file in portable mode
62 // On non-portable (registry on Windows), we cannot safely backup
63 QString portable_ini =
64 QCoreApplication::applicationDirPath() + QDir::separator() + "qtpass.ini";
65 m_isPortableMode = QFile::exists(portable_ini);
66
67 if (m_isPortableMode) {
68 QtPassSettings::getInstance()->sync();
69 QString settingsFile = QtPassSettings::getInstance()->fileName();
70 m_settingsBackupPath = settingsFile + ".bak";
71 QFile::remove(m_settingsBackupPath);
72 QFile::copy(settingsFile, m_settingsBackupPath);
73 } else {
74 m_settingsBackupPath.clear();
75 // On non-portable mode, warn but continue (tests may modify registry)
76 qWarning()
77 << "Non-portable mode detected. Tests may modify registry settings.";
78 }
79}

81void tst_settings::cleanupTestCase() {
82 // Restore original settings after all tests
83 // This ensures make check doesn't change user's live config
84 if (m_isPortableMode && !m_settingsBackupPath.isEmpty()) {
85 QString settingsFile = QtPassSettings::getInstance()->fileName();
86 QFile::remove(settingsFile);
87 QFile::copy(m_settingsBackupPath, settingsFile);
88 QFile::remove(m_settingsBackupPath);
89 }

218void tst_settings::boolRoundTrip_data() {
219 QTest::addColumn<QString>("setting");
220 QTest::addColumn<bool>("testValue");
221
222 for (const auto &s : boolSettings) {
223 QByteArray name(s.name);
224 QTest::newRow(name + "_true") << s.name << true;
225 QTest::newRow(name + "_false") << s.name << false;
226 }
227}

229void tst_settings::boolRoundTrip() {
230 QFETCH(QString, setting);
231 QFETCH(bool, testValue);
232
233 for (const auto &s : boolSettings) {
234 if (setting == s.name) {
235 s.setter(testValue);
236 QVERIFY2(s.getter(testValue) == testValue,
237 qPrintable(QString("%1 should be %2, got %3")
238 .arg(setting)
239 .arg(testValue ? "true" : "false")
240 .arg(s.getter(testValue) ? "true" : "false")));
241 return;
242 }
243 }
244 QFAIL(qPrintable(QString("Unknown setting: %1").arg(setting)));
245}

264void tst_settings::autoDetectGit() {
265 QTemporaryDir tempDir;
266 QtPassSettings::setPassStore(tempDir.path());
267
268 QDir gitDir(tempDir.path());
269 QVERIFY(gitDir.mkdir(".git"));
270 QtPassSettings::getInstance()->sync();
271
272 QtPassSettings::getInstance()->remove("useGit");
273 QtPassSettings::getInstance()->sync();
274 QVERIFY2(QtPassSettings::isUseGit(true),
275 "Should auto-detect .git and return true when default is true");
276
277 QtPassSettings::getInstance()->remove("useGit");
278 QtPassSettings::getInstance()->sync();
279 QVERIFY2(!QtPassSettings::isUseGit(false),
280 "Should respect false default and not auto-detect when .git exists");
281
282 QVERIFY(gitDir.rmdir(".git"));
283 QtPassSettings::getInstance()->sync();
284
285 QtPassSettings::getInstance()->remove("useGit");
286 QtPassSettings::getInstance()->sync();
287 QVERIFY2(QtPassSettings::isUseGit(true),
288 "Should return true default when .git not present");
289
290 QtPassSettings::getInstance()->remove("useGit");
291 QtPassSettings::getInstance()->sync();
292 QVERIFY2(!QtPassSettings::isUseGit(false),
293 "Should return false default when .git not present");
294}
