245 QVERIFY2(captured.contains("?"), "URL should include query params");
246 QVERIFY2(captured.contains("#"), "URL should include fragment");

559void tst_util::boundedRandom() {
560 ImitatePass pass;
561
562 QVector<quint32> counts(10, 0);
563 const int iterations = 1000;
564
565 for (int i = 0; i < iterations; ++i) {
566 QString result = pass.Generate_b(1, "0123456789");
567 quint32 val = result.at(0).digitValue();
568 if (val < 10) {
569 counts[val]++;
570 }
571 }
572
573 for (int i = 0; i < 10; ++i) {
574 QVERIFY2(counts[i] > 0, "Each digit should appear at least once");
