582void tst_util::boundedRandom() {
583 ImitatePass pass;
584
585 QVector<quint32> counts(10, 0);
586 const int iterations = 1000;
587
588 for (int i = 0; i < iterations; ++i) {
589 QString result = pass.generatePassword(1, "0123456789");
590 quint32 val = result.at(0).digitValue();
591 QVERIFY2(val < 10, "generatePassword should only return digit characters");
592 counts[val]++;
593 }
594
595 for (int i = 0; i < 10; ++i) {
596 QVERIFY2(counts[i] > 0, "Each digit should appear at least once");
597 }
598
599 const double expected = static_cast<double>(iterations) / 10.0;
600 double chi2 = 0.0;
601 for (int i = 0; i < 10; ++i) {
602 const double count = static_cast<double>(counts[i]);
603 chi2 += (count - expected) * (count - expected) / expected;
604 }
605 const double chi2Critical = 25.0;
606 QVERIFY2(chi2 < chi2Critical,
607 qPrintable(
608 QStringLiteral("Chi-square %1 exceeds critical value %2 (df=9)")
609 .arg(chi2)
610 .arg(chi2Critical)));
611}
